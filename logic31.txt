#include "vars.txt"

LOGIC__HEADER

#define oBOWL o1

if (fROOM_SCRIPT_EXECUTED) {
  load.pic(vROOM_NO);
  draw.pic(vROOM_NO);
  discard.pic(vROOM_NO);
  load.sound(SND_OUCH);
  load.sound(SND_FALLING);
  set.horizon(50);
  if (fGOAT_COMPANION) {
    load.logics(LOGIC103_GOAT);
  }
  if (vROOM_YOU_KILLED_THE_GOAT_IN == vROOM_NO) {
    add.to.pic(62, 0, 0, 39, 96, 0, 0);
  }
  if (fCARRYING_MAGIC_BEANS || vBEANSTALK_ROOM == ROOM31_FREE_BOWL) {
    vTEMP_PIC = 84;
    set(f155);
    load.pic(vTEMP_PIC);
    load.view(114);
    load.view(103);
    load.view(104);
    load.view(74);
    load.view(105);
    load.logics(LOGIC104_BEANSTALK);
    load.sound(SND_BEANSTALK);
  }
  if (vPREV_ROOM_NO == ROOM70 && fEGO_CLIMBING_BEANSTALK) {
    set.view(oEGO, 74);
    position(oEGO, 64, 51);
    move.obj(oEGO, 64, 51, 0, f148);
    set.priority(oEGO, 8);
  }
  if (vPREV_ROOM_NO == ROOM70 && fFALLING_TO_EARTH) {
    program.control();
    ignore.horizon(oEGO);
    set.priority(oEGO, 15);
    set.view(oEGO, 105);
    get.posn(oEGO, vTEMP_X, vTEMP_Y);
    if (vTEMP_X > 75) {
      position(oEGO, 80, 29);
    }
    else {
      position(oEGO, 65, 29);
    }
    get.posn(oEGO, vTEMP_X, vTEMP_Y);
    vTEMP_Y = 106;
    vTEMP_STEP = 2;
    move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, rf2);
  }
  if (!fBOWL_TAKEN) {
    load.view(25);
    animate.obj(oBOWL);
    set.view(oBOWL, 25);
    position(oBOWL, 126, 138);
  }
  if (!fBOWL_TAKEN) {
    draw(oBOWL);
  }
  if (vBEANSTALK_ROOM == ROOM31_FREE_BOWL) {
    overlay.pic(vTEMP_PIC);
  }
  draw(oEGO);
  show.pic();
}
if (fFALLING_TO_EARTH && !rf3) {
  set(rf3);
  set(f135);
  sound(SND_FALLING, rf4);
}
if (!rf2 && rf4) {
  reset(rf3);
  reset(rf4);
}
if (rf2) {
  stop.sound();
  set(rf3);
  reset(fFALLING_TO_EARTH);
  reset(f135);
  set.view(oEGO, 114);
  shake.screen(2);
  set(fGAME_OVER);
}
if (fBOWL_TAKEN && !fCARRYING_BOWL) {
  if (said("anyword", "bowl")) {
    print("There may have been a bowl here at one time, but it is not here now!");
  }
}
if (!fBOWL_TAKEN) {
  if (said("examine", "bowl")) {
    if (posn(oEGO, 106, 103, 159, 167)) {
      print("This is a very colorful ceramic bowl.");
    }
    else {
      print("It looks like an interesting, colorful bowl.");
    }
  }
  if (said("examine", "ground")) {
    print("There appears to be some sort of bowl on the ground.");
  }
}
if (!fBOWL_TAKEN && said("get", "bowl")) {
  if (posn(oEGO, 116, 128, 136, 148)) {
    print("You pick up the ceramic bowl.");
    start.update(oBOWL);
    erase(oBOWL);
    vSCORE += 3;
    set(fCARRYING_BOWL);
    set(fBOWL_TAKEN);
    get("ceramic bowl");
  }
  else {
    print("That is impossible from here.");
  }
}
if (fCARRYING_MAGIC_BEANS && f155) {
  if (said("plant", "magic beans")) {
    reset(fCARRYING_MAGIC_BEANS);
    vBEANSTALK_ROOM = ROOM31_FREE_BOWL;
    set(fMAGIC_BEANS_PLANTED);
    drop("beans");
    vSCORE += 2;
    print("The beans immediately sprout roots that reach deep into the fertile soil... leaves unfold themselves to the warm, loving sun... a rumbling is felt... and a mighty beanstalk now stretches up into the sky!");
    sound(SND_BEANSTALK, rf5);
    overlay.pic(vTEMP_PIC);
    show.pic();
  }
}
if (vBEANSTALK_ROOM == ROOM31_FREE_BOWL && !fEGO_INVISIBLE) {
  if (said("climb", "beanstalk") && !fEGO_CLIMBING_BEANSTALK) {
    if (posn(oEGO, 52, 76, 103, 97)) {
      erase(oEGO);
      set.view(oEGO, 74);
      set(fEGO_CLIMBING_BEANSTALK);
      position(oEGO, 71, 77);
      set.priority(oEGO, 8);
      draw(oEGO);
      move.obj(oEGO, 71, 77, 0, f148);
    }
    else {
      print("You can't reach the beanstalk from here.");
    }
  }
}
if (vBEANSTALK_ROOM != ROOM31_FREE_BOWL) {
  if (said("climb", "beanstalk")) {
    print("There is no beanstalk here to climb.");
  }
}
if (fEGO_CLIMBING_BEANSTALK) {
  if (fEGO_TOUCHED_TRIGGER_LINE && posn(oEGO, 60, 78, 93, 80)) {
    erase(oEGO);
    set.view(oEGO, 0);
    reset(fEGO_CLIMBING_BEANSTALK);
    position(oEGO, 70, 106);
    release.priority(oEGO);
    draw(oEGO);
  }
  if (fEGO_TOUCHED_TRIGGER_LINE && !posn(oEGO, 60, 78, 93, 80)) {
    program.control();
    set.priority(oEGO, 15);
    get.posn(oEGO, vTEMP_X, vTEMP_Y);
    erase(oEGO);
    set.view(oEGO, 105);
    reset(fEGO_CLIMBING_BEANSTALK);
    position.v(oEGO, vTEMP_X, vTEMP_Y);
    draw(oEGO);
    vTEMP_Y = 106;
    vTEMP_STEP = 2;
    move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, rf0);
    set(f135);
    sound(SND_FALLING, f198);
  }
}
rv0--;
if (rf0) {
  stop.sound();
  reset(rf0);
  observe.blocks(oEGO);
  release.priority(oEGO);
  stop.motion(oEGO);
  set.view(oEGO, 103);
  rv0 = 20;
  set(f98);
  shake.screen(2);
  print("Ouch!");
  set(f135);
  sound(SND_OUCH, fSOUND_END_MARKER);
}
if (fSOUND_END_MARKER) {
  if (rv0 != 0) {
    reset(fSOUND_END_MARKER);
    sound(SND_OUCH, fSOUND_END_MARKER);
  }
}
if (rv0 == 1) {
  set.view(oEGO, 104);
  end.of.loop(oEGO, rf1);
}
if (rf1) {
  reset(rf1);
  reset(f98);
  start.motion(oEGO);
  set.view(oEGO, 0);
  release.priority(oEGO);
  player.control();
}
if (vEGO_EDGE_CODE != EDGE_NONE) {
  reset(f155);
}
if (vEGO_EDGE_CODE == EDGE_HORIZON && fEGO_CLIMBING_BEANSTALK) {
  new.room(ROOM70);
}
if (vEGO_EDGE_CODE == EDGE_HORIZON && !fEGO_CLIMBING_BEANSTALK) {
  new.room(ROOM34);
}
if (vEGO_EDGE_CODE == EDGE_RIGHT) {
  new.room(ROOM32);
}
if (vEGO_EDGE_CODE == EDGE_LEFT) {
  new.room(ROOM30_WALNUT_TREE);
}
if (vEGO_EDGE_CODE == EDGE_BOTTOM) {
  new.room(ROOM18_CLIFFS_LAKE_WEST);
}
if (fGOAT_COMPANION) {
  call(LOGIC103_GOAT);
}
if (vBEANSTALK_ROOM == ROOM31_FREE_BOWL) {
  call(LOGIC104_BEANSTALK);
}
return;
