if (fROOM_SCRIPT_EXECUTED) {
  load.pic(vROOM_NO);
  draw.pic(vROOM_NO);
  discard.pic(vROOM_NO);
  load.view(105);
  load.view(74);
  load.sound(SND_FALLING);
  if (vPREV_ROOM_NO == ROOM72 && fEGO_CLIMBING_BEANSTALK) {
    set.view(oEGO, 74);
    position(oEGO, 66, 40);
    move.obj(oEGO, 66, 40, 0, f148);
  }
  if (vPREV_ROOM_NO == ROOM72 && f158) {
    program.control();
    set.view(oEGO, 105);
    get.posn(oEGO, vTEMP_X, vTEMP_Y);
    if (vTEMP_X > 75) {
      position(oEGO, 90, 29);
    }
    else {
      position(oEGO, 55, 29);
    }
    vTEMP_Y = 166;
    vTEMP_STEP = 4;
    set.priority(oEGO, 15);
    move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, f20);
  }
  if (vPREV_ROOM_NO == ROOM70) {
    set.view(oEGO, 74);
    position(oEGO, 69, 160);
    move.obj(oEGO, 69, 160, 0, f148);
  }
  draw(oEGO);
  show.pic();
}
if (f158 && !f20 && !f21) {
  sound(SND_FALLING, fSOUND_END_MARKER);
  set(f21);
}
if (fSOUND_END_MARKER) {
  if (!f20) {
    reset(f21);
    reset(fSOUND_END_MARKER);
  }
}
if (said("examine", "beanstalk")) {
  print("You can no longer see the bottom... it seems to go  on forever!");
}
if (fEGO_TOUCHED_TRIGGER_LINE && !f158) {
  program.control();
  print("You have to be careful when you get so high on the foliage!");
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  reset(fEGO_CLIMBING_BEANSTALK);
  set(f158);
  erase(oEGO);
  set.view(oEGO, 105);
  if (vTEMP_X > 75) {
    vTEMP_X = 80;
    position.v(oEGO, vTEMP_X, vTEMP_Y);
    draw(oEGO);
  }
  if (vTEMP_X < 76) {
    vTEMP_X = 65;
    position.v(oEGO, vTEMP_X, vTEMP_Y);
    draw(oEGO);
  }
  start.cycling(oEGO);
  set(f98);
  vTEMP_Y = 166;
  vTEMP_STEP = 4;
  move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, f20);
}
if (vEGO_EDGE_CODE == EDGE_HORIZON) {
  new.room(ROOM72);
}
if (f20 || posn(oEGO, 0, 166, 159, 167)) {
  new.room(ROOM70);
}
return;
