#include "vars.txt"

LOGIC__HEADER

#define oGOAT o13

if (fROOM_SCRIPT_EXECUTED) {
  load.view(14);
  animate.obj(oGOAT);
  set.view(oGOAT, 14);
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  object.on.land(oGOAT);
  if (vROOM_NO == ROOM7_WATERFALL_LAKE_EAST && vPREV_ROOM_NO == ROOM8_WATERFALL_LAKE_WEST && vTEMP_Y < 65) {
    position(oGOAT, 1, 42);
    stop.motion(oGOAT);
    stop.cycling(oGOAT);
    ignore.horizon(oGOAT);
    ignore.blocks(oGOAT);
    set(rf20);
    draw(oGOAT);
    goto(ATRIVER);
  }
  if (vROOM_NO == ROOM17_CLIFFS_LAKE_EAST && vPREV_ROOM_NO == ROOM18_CLIFFS_LAKE_WEST && vTEMP_Y < 65) {
    position(oGOAT, 1, 45);
    stop.motion(oGOAT);
    stop.cycling(oGOAT);
    ignore.horizon(oGOAT);
    ignore.blocks(oGOAT);
    set(rf23);
    draw(oGOAT);
    goto(ATRIVER);
  }
  if (vROOM_NO == ROOM17_CLIFFS_LAKE_EAST && vPREV_ROOM_NO == ROOM32 && vTEMP_X < 50) {
    position(oGOAT, 1, 45);
    stop.motion(oGOAT);
    stop.cycling(oGOAT);
    ignore.horizon(oGOAT);
    ignore.blocks(oGOAT);
    set(rf23);
    draw(oGOAT);
    goto(ATRIVER);
  }
  if (vTEMP_Y > 100) {
    vTEMP_Y += 1;
  }
  else {
    vTEMP_Y -= 1;
  }
  position.v(oGOAT, vTEMP_X, vTEMP_Y);
  v107 = 1;
  step.size(oGOAT, v107);
  draw(oGOAT);
  follow.ego(oGOAT, 10, rf07);
}
distance(oEGO, oGOAT, rv10);
if (rv10 > 35) {
  v107 = 2;
  step.size(oGOAT, v107);
}
else {
  v107 = 1;
  step.size(oGOAT, v107);
}
if (!fGOAT_COMPANION) {
  return;
}
if (fGAME_OVER) {
  normal.motion(oGOAT);
  stop.motion(oGOAT);
  stop.cycling(oGOAT);
  return;
}
if (rf20 || rf23) {
  goto(ATRIVER);
}
if (f76) {
  ignore.blocks(oGOAT);
  return;
}
observe.blocks(oGOAT);
rv11--;
if (rv11 == 1) {
  if (rv10 < 21) {
    set(rf07);
  }
  else if (!rf07 && vAQUATIC_ACTIVITY == AQUATICACTIVITY_NONE) {
    rv11 = 0;
    start.cycling(oGOAT);
    start.motion(oGOAT);
    follow.ego(oGOAT, 10, rf07);
  }
}
if (rf07) {
  reset(rf07);
  rv11 = 9;
  normal.motion(oGOAT);
  stop.motion(oGOAT);
  stop.cycling(oGOAT);
}
if (rv10 > 20 && vAQUATIC_ACTIVITY != AQUATICACTIVITY_NONE && !fGOAT_WONT_SWIM) {
  if (vROOM_NO != ROOM1_CASTLE_EAST
    && vROOM_NO != ROOM2_CASTLE_WEST
    && vROOM_NO != ROOM41
    && vROOM_NO != ROOM33
    && vROOM_NO != ROOM34
    && vROOM_NO != ROOM32
    && vROOM_NO != ROOM25
    && vROOM_NO != ROOM26
    && vROOM_NO != ROOM39
  ) {
    print("The goat does not mind getting its feet wet,but will not swim in the water.  You should go back and lead the goat around the water.");
  }
  set(fGOAT_WONT_SWIM);
  start.cycling(oGOAT);
  wander(oGOAT);
}
if (fGOAT_WONT_SWIM && rv10 < 25 && vAQUATIC_ACTIVITY == AQUATICACTIVITY_NONE) {
  reset(fGOAT_WONT_SWIM);
  follow.ego(oGOAT, 10, rf07);
}
if (fEGO_INVISIBLE && !rf22) {
  set(rf22);
  print("The goat, not being able to see you, wanders off.");
  reset(fGOAT_COMPANION);
  wander(oGOAT);
  if (vOBJECT_EDGE_CODE != 0) {
    erase(oGOAT);
  }
}
return;
ATRIVER:
  distance(oEGO, oGOAT, rv10);
  if (rf23 && vAQUATIC_ACTIVITY != AQUATICACTIVITY_NONE && !fGOAT_WONT_SWIM) {
    set(fGOAT_WONT_SWIM);
    print("The goat does not mind getting its feet wet,but will not swim in the water.  You should go back and lead the goat around the water.");
  }
  if (rf23 && fGOAT_WONT_SWIM && rv10 < 25 && vAQUATIC_ACTIVITY == AQUATICACTIVITY_NONE) {
    reset(fGOAT_WONT_SWIM);
  }
  return;

LOGIC__FOOTER
