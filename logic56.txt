#include "vars.txt"

LOGIC__HEADER

if (fROOM_SCRIPT_EXECUTED) {
  ignore.horizon(oEGO);
  load.pic(vROOM_NO);
  draw.pic(vROOM_NO);
  discard.pic(vROOM_NO);
  load.sound(SND_FALLING);
  load.view(105);
  if (vPREV_ROOM_NO == ROOM57) {
    get.posn(oEGO, vTEMP_X, vTEMP_Y);
    if (vTEMP_Y > 154) {
      position(oEGO, 150, 153);
    }
  }
  draw(oEGO);
  show.pic();
}
if (fEGO_TOUCHED_TRIGGER_LINE && posn(oEGO, 0, 0, 159, 83)) {
  set(f199);
  reset(fEGO_TOUCHED_TRIGGER_LINE);
  program.control();
  if (!fEGO_INVISIBLE) {
    set.view(oEGO, 105);
  }
  start.cycling(oEGO);
  set(f98);
  get.priority(oEGO, v30);
  set.priority.v(oEGO, v30);
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  vTEMP_Y += 40;
  vTEMP_STEP = 4;
  move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, f20);
}
if (fEGO_TOUCHED_TRIGGER_LINE && posn(oEGO, 0, 84, 159, 167)) {
  set(f199);
  set(f95);
  program.control();
  stop.cycling(oEGO);
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  get.priority(oEGO, v30);
  set.priority.v(oEGO, v30);
  vTEMP_Y += 9;
  vTEMP_STEP = 4;
  move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, f20);
}
if (!f20 && f199 && !f21) {
  sound(SND_FALLING, fSOUND_END_MARKER);
  set(f21);
}
if (fSOUND_END_MARKER) {
  if (!f20) {
    reset(fSOUND_END_MARKER);
    reset(f21);
  }
}
if (f20) {
  erase(oEGO);
  reset(f199);
  reset(f20);
  stop.sound();
  print("You should know you can't walk on clouds.");
  set(fGAME_OVER);
}
if (said("examine", "flowers")) {
  print("These flowers look different than any you have ever seen before.");
}
else if (said("examine", "anyword")) {
  print("You are in the Land of the Clouds.  It is rumored that a giant lives up here.");
}
if (vEGO_EDGE_CODE == EDGE_RIGHT) {
  new.room(ROOM57);
}
if (vEGO_EDGE_CODE == EDGE_LEFT) {
  if (fMAGIC_BEANS_PLANTED) {
    new.room(ROOM72);
  }
  else {
    new.room(ROOM64);
  }
}
return;

LOGIC__FOOTER
