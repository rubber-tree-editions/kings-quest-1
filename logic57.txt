#include "vars.txt"

LOGIC__HEADER

#define rfFELL_THROUGH_CLOUDS rf0
#define rfPLAYING_SND_FALLING rf1

#define rvLOCK_PRIORITY rv0

#if RUBBER_TREE
#define rvRESTORE_EGO_X rv1
#define rvRESTORE_EGO_Y rv2
#define rvRESTORE_EGO_DIR rv3
#endif

#if RUBBER_TREE
if (fUNDO && fGAME_OVER) {
  stop.sound();
  erase(oEGO);
  position.v(oEGO, rvRESTORE_EGO_X, rvRESTORE_EGO_Y);
  reset(fCYCLE_EGO_ANIMATION);
  stop.cycling(oEGO);
  set.view(oEGO, 0);
  set.dir(oEGO, rvRESTORE_EGO_DIR);
  release.priority(oEGO);
  draw(oEGO);
  reset(fUNDO);
  reset(fGAME_OVER);
  reset(fFALLING);
  player.control();
  return;
}
#endif
if (fROOM_SCRIPT_EXECUTED) {
  ignore.horizon(oEGO);
  load.pic(vROOM_NO);
  draw.pic(vROOM_NO);
  discard.pic(vROOM_NO);
  load.sound(SND_FALLING);
  load.view(105);
  draw(oEGO);
  show.pic();
}
if (fEGO_TOUCHED_TRIGGER_LINE) {
  #if RUBBER_TREE
    get.posn(oEGO, rvRESTORE_EGO_X, rvRESTORE_EGO_Y);
    rvRESTORE_EGO_Y++;
    rvRESTORE_EGO_DIR = vEGO_DIR;
  #endif
  #if VERSION_S
    prevent.input();
    set(f215);
    set(fFALLING);
    ignore.blocks(oEGO);
  #else
    ignore.blocks(oEGO);
    set(fFALLING);
  #endif
  reset(fEGO_TOUCHED_TRIGGER_LINE);
  program.control();
  if (!fEGO_INVISIBLE) {
    set.view(oEGO, 105);
  }
  start.cycling(oEGO);
  set(fCYCLE_EGO_ANIMATION);
  get.priority(oEGO, rvLOCK_PRIORITY);
  set.priority.v(oEGO, rvLOCK_PRIORITY);
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  vTEMP_Y += 30;
  vTEMP_STEP = 4;
  move.obj.v(oEGO, vTEMP_X, vTEMP_Y, vTEMP_STEP, rfFELL_THROUGH_CLOUDS);
}
if (!rfFELL_THROUGH_CLOUDS && fFALLING && !rfPLAYING_SND_FALLING) {
  sound(SND_FALLING, fSOUND_END_MARKER);
  set(rfPLAYING_SND_FALLING);
}
#if VERSION_U || VERSION_V
if (fSOUND_END_MARKER) {
  if (!rfFELL_THROUGH_CLOUDS) {
    reset(fSOUND_END_MARKER);
    reset(rfPLAYING_SND_FALLING);
  }
}
#endif
if (rfFELL_THROUGH_CLOUDS) {
  erase(oEGO);
  reset(fFALLING);
  reset(rfFELL_THROUGH_CLOUDS);
  stop.sound();
  print("You'll have to keep yourself out of the clouds!");
  set(fGAME_OVER);
  #if VERSION_S
    accept.input();
  #endif
}
if (said("examine", "tree")) {
  print("These are very unusual looking trees.");
}
else if (said("examine", "anyword")) {
  print("You are in the Land of the Clouds.  It is rumored that a giant lives up here.");
}
if (vEGO_EDGE_CODE == EDGE_RIGHT) {
  new.room(ROOM58_GIANT_STOMPING_GROUND);
}
if (vEGO_EDGE_CODE == EDGE_LEFT) {
  new.room(ROOM56_CLOUDS_GRASS_EDGE);
}
if (vEGO_EDGE_CODE == EDGE_BOTTOM) {
  new.room(ROOM60_CLOUDLANDS_SOUTHWEST);
}
#if VERSION_S
  get.posn(oEGO, vTEMP_X, vTEMP_Y);
  if (vTEMP_Y < 60) {
    set(f216);
  }
  else {
    reset(f216);
  }
#endif
return;

LOGIC__FOOTER
